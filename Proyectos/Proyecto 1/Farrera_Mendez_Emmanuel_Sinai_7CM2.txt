arreglo resultante. Ejemplo, tenemos el siguiente arreglo 1,2,3,4,5, si desplazamos los elementos una
posici√≥n hacia adelante debemos obtener: 5,1,2,3,4. Es decir, el primer elemento se mueve hacia el
segundo lugar, el segundo al tercero, etc., y el √∫ltimo al primero.
*/

import java.util.Scanner;


public class proyecto1 {
    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);

        int[] arregloNumeros = new int[10];

        System.out.println("Ingresa 10 n√∫meros:");
        for (int i = 0; i < 10; i++) {
            arregloNumeros[i] = scanner.nextInt();
        }

        System.out.println("\nLos n√∫meros introducidos son:");
        for (int j = 0; j < arregloNumeros.length; j++) {
            System.out.print(arregloNumeros[j] + "\t");
        }
        System.out.println();

       
        int ultimoElemento = arregloNumeros[arregloNumeros.length - 1];

        
        for (int k = arregloNumeros.length - 1; k > 0; k--) {
            arregloNumeros[k] = arregloNumeros[k - 1];
        }

        
        arregloNumeros[0] = ultimoElemento;

       
        System.out.println("\nEl arreglo resultante despu√©s de mover los elementos una posici√≥n hacia adelante:");
        for (int y = 0; y < arregloNumeros.length; y++) {
            System.out.print(arregloNumeros[y] + "\t");
        }
        System.out.println();

        scanner.close();
        
    }
}

/* 
23 Escribe un programa que solicite edades y estaturas de personas y guarde los datos en un
archivo de texto de la siguiente manera: edad, estatura. El programa debe crear el archivo en caso de
que este no exista. En caso de que el archivo ya exista entonces deber√° a√±adir los datos al archivo. El
programa debe terminar cuando el usuario introduzca el n√∫mero 0 como edad.
*/

import java.util.Scanner;
import java.util.Random;
import java.io.BufferedReader;
import java.io.File;
import java.io.FileInputStream;
import java.io.InputStreamReader;


public class proyecto1 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        try {
            
            File archivo = new File("datos_personas.txt");
            FileWriter fw = new FileWriter(archivo, true); 
            BufferedWriter bw = new BufferedWriter(fw);

           
            while (true) {
                System.out.println("Ingrese la edad de la persona (0 para salir):");
                int edad = scanner.nextInt();
                if (edad == 0) {
                    break;
                }
                System.out.println("Ingrese la estatura de la persona:");
                double estatura = scanner.nextDouble();

               
                bw.write(edad + ", " + estatura);
                bw.newLine();
            }

            
            bw.close();
            System.out.println("Datos guardados en el archivo.");
        } catch (IOException e) {
            System.out.println("Error al manipular el archivo: " + e.getMessage());
        } finally {
            scanner.close();
        }
    }
}

/* 
24 Leer un archivo de texto que contiene un n√∫mero entero por cada l√≠nea e indicar cu√°l es el
n√∫mero mayor dentro del archivo y el n√∫mero de rengl√≥n en que se encuentra.
*/

import java.io.BufferedReader;
import java.io.File;
import java.io.FileReader;
import java.io.IOException;


public class proyecto1 {
    public static void main(String[] args) {
        File archivo = new File("numeros.txt");

        int numeroMayor = Integer.MIN_VALUE; 
        int numeroRenglon = -1;
        int renglonActual = 0;

        try (BufferedReader br = new BufferedReader(new FileReader(archivo))) {
            String linea;

            while ((linea = br.readLine()) != null) {
                renglonActual++;

                
                int numero = Integer.parseInt(linea);

                
                if (numero > numeroMayor) {
                    numeroMayor = numero;
                    numeroRenglon = renglonActual;
                }
            }
        } catch (IOException e) {
            System.out.println("Error al leer el archivo: " + e.getMessage());
        } catch (NumberFormatException e) {
            System.out.println("El archivo contiene datos no num√©ricos.");
        }

        if (numeroRenglon != -1) {
            System.out.println("El n√∫mero mayor es " + numeroMayor + " y se encuentra en el rengl√≥n " + numeroRenglon + ".");
        } else {
            System.out.println("El archivo est√° vac√≠o o no contiene n√∫meros.");
        }
    }
}
